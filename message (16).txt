    Config.Positions[k] = {X = v.X, Y = v.Y}

if isfile and isfile(FileName) then
    pcall(function()
        local ok, decoded = pcall(function()
            return HttpService:JSONDecode(readfile(FileName))
        end)
        if not ok then return end
        -- top-level keys
        for k, v in pairs(DefaultConfig) do
            if k ~= "Positions" and k ~= "AnimalConfigs" and decoded[k] == nil then
                decoded[k] = v
            end
        end
        -- positions
        if decoded.Positions then
            for k, v in pairs(DefaultConfig.Positions) do
                if decoded.Positions[k] == nil then decoded.Positions[k] = {X=v.X, Y=v.Y} end
            end
        else
            decoded.Positions = {}
            for k, v in pairs(DefaultConfig.Positions) do
                decoded.Positions[k] = {X=v.X, Y=v.Y}
            end
        end
        -- animal configs
        if not decoded.AnimalConfigs then decoded.AnimalConfigs = {} end
        Config = decoded
    end)
end

local function SaveConfig()
    if writefile then
        pcall(function()
            writefile(FileName, HttpService:JSONEncode(Config))
        end)
    end
end

-- // ================== JOB ID DECRYPTOR ================== \\ --

local function xorDecrypt(encrypted, key)
    local decrypted = ""
    local bxor = bit32 and bit32.bxor or (bit and bit.bxor) or function(a, b) return (a + b) % 256 end
    for i = 1, #encrypted do
        local charCode = string.byte(encrypted, i)
        local keyIndex = ((i - 1) % #key) + 1
        local keyChar  = string.byte(key, keyIndex)
        decrypted = decrypted .. string.char(bxor(charCode, keyChar))
    end
    return decrypted
end

local function fromBase64(data)
    local b = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/'
    data = string.gsub(data, '[^'..b..'=]', '')
    return (data:gsub('.', function(x)
        if (x == '=') then return '' end
        local r, f = '', (b:find(x) - 1)
        for i = 6, 1, -1 do r = r .. (f % 2^i - f % 2^(i-1) > 0 and '1' or '0') end
        return r
    end):gsub('%d%d%d?%d?%d?%d?%d?%d?', function(x)
        if (#x ~= 8) then return '' end
        local c = 0
        for i = 1, 8 do c = c + (x:sub(i,i) == '1' and 2^(8-i) or 0) end
        return string.char(c)
    end))
end

local function generateKey(baseKey, salt)
    local combined = baseKey .. salt
    local key = ""
    for i = 1, 24 do
        local index = ((i - 1) % #combined) + 1
        key = key .. string.char((string.byte(combined, index) * (i + 13)) % 256)
    end
    return key
end

local function decryptJobId(encryptedText)
    local SECRET_KEY = "AutoJoinerSecret2024"
    local SALT       = "StrawberryBerry123"
    if not encryptedText or encryptedText == "" then return encryptedText end
    if not string.match(encryptedText, "^[A-Za-z0-9+/=]+$") then return encryptedText end
    if #encryptedText < 20 then return encryptedText end
    local decoded = fromBase64(encryptedText)
    if not decoded or decoded == "" then return encryptedText end
    local key       = generateKey(SECRET_KEY, SALT)
    local decrypted = xorDecrypt(decoded, key)
    if #decrypted >= 8 then
        local prefix   = decrypted:sub(1, 4)
        local saltPart = decrypted:sub(5, 8)
        if prefix == "STB_" and saltPart == "Stra" then return decrypted:sub(9) end
    end
    local stbPos = decrypted:find("STB_")
    if stbPos and #decrypted >= stbPos + 8 then return decrypted:sub(stbPos + 8) end
    local clean = decrypted:gsub("%s+", ""):gsub("%c+", "")
    if string.match(clean, "^[0-9a-f]+$") and #clean >= 60 and #clean <= 70 then return clean end
    return encryptedText
end

-- // ================== TELEPORT FUNKTION ================== \\ --

local function TryTeleport(jobId, retryCount)
    if not jobId or jobId == "" then return end
    retryCount = retryCount or 1
    task.spawn(function()
        for i = 1, retryCount do
            pcall(function()
                TeleportService:TeleportToPlaceInstance(game.PlaceId, jobId, player)
            end)
            if i < retryCount then task.wait(1.5) end
        end
    end)
end

-- Position-Tracking via Heartbeat:
-- Erkennt wenn sich eine UI bewegt hat und speichert automatisch (Debounce 1s)
local _lastSB = {X = -1, Y = -1}
local _lastMF = {X = -1, Y = -1}
local _saveTimer = 0
RunService.Heartbeat:Connect(function(dt)
    _saveTimer = _saveTimer + dt
    if _saveTimer < 0.5 then return end
    _saveTimer = 0

    local vp   = workspace.CurrentCamera.ViewportSize
    local sbAP = StatusBar.AbsolutePosition
    local mfAP = MainFrame.AbsolutePosition
    local sbX  = math.floor(sbAP.X * 100) / 100
    local sbY  = math.floor(sbAP.Y * 100) / 100
    local mfX  = math.floor(mfAP.X * 100) / 100
    local mfY  = math.floor(mfAP.Y * 100) / 100

    if sbX ~= _lastSB.X or sbY ~= _lastSB.Y or
       mfX ~= _lastMF.X or mfY ~= _lastMF.Y then
        _lastSB = {X = sbX, Y = sbY}
        _lastMF = {X = mfX, Y = mfY}
        Config.Positions.StatusBar = {X = sbAP.X / vp.X, Y = sbAP.Y / vp.Y}
        Config.Positions.MainFrame = {X = mfAP.X / vp.X, Y = mfAP.Y / vp.Y}
        SaveConfig()
    end
end)

-- // FLOOR UNLOCK
local UNLOCK_LEVELS = {-2, 15, 32}

local function triggerClosestUnlock(yLevel)
    local character = player.Character
    local hrp = character and character:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    local playerY    = yLevel or hrp.Position.Y
    local Y_THRESHOLD = 5
    local bestSameLevel, distSame = nil, math.huge
    local bestFallback,  distFall = nil, math.huge
    local plots = workspace:FindFirstChild("Plots")
    if not plots then return end
    for _, obj in ipairs(plots:GetDescendants()) do
        if obj:IsA("ProximityPrompt") and obj.Enabled then
            local part = obj.Parent
            if part and part:IsA("BasePart") then
                local dist  = (hrp.Position - part.Position).Magnitude
                local yDiff = math.abs(playerY - part.Position.Y)
                if dist < distFall then distFall = dist; bestFallback = obj end
                if yDiff <= Y_THRESHOLD and dist < distSame then
                    distSame = dist; bestSameLevel = obj
                end
            end
        end
    end
    local target = bestSameLevel or bestFallback
    if not target then return end
    if fireproximityprompt then
        fireproximityprompt(target)
    else
        pcall(function() target:InputBegan(Enum.UserInputType.MouseButton1) end)
        task.wait(0.05)
        pcall(function() target:InputEnded(Enum.UserInputType.MouseButton1) end)
    end
end

-- ===== GLOBALES SPAM SYSTEM =====
local globalSpam = {
    active    = false,
    thread    = nil,
    spamBtnRef = nil,
    animalName = "",
}
autoSpamTasks = {}

local function StartAutoSpam(animalName, msValue, retryCount)
    if retryCount < 1 then retryCount = 1 end
    local taskId      = animalName .. "_" .. tostring(os.clock())
    local consoleLabel = AddConsoleEntry(animalName, msValue, 1, retryCount)
    local thread
    thread = task.spawn(function()
        autoSpamTasks[taskId] = thread
        for i = 1, retryCount do
            if autoSpamTasks[taskId] == nil then break end
            if consoleLabel and consoleLabel.Parent then
                consoleLabel.Text = "[" .. os.date("%H:%M") .. "] [" .. i .. "/" .. retryCount .. "] " .. animalName .. " $" .. msValue .. "M/s"
            end
            task.wait(1.5)
        end
        autoSpamTasks[taskId] = nil
    end)
end

StopCurrentSpam = function()
    if globalSpam.active then
        globalSpam.active = false
        if globalSpam.thread then
            task.cancel(globalSpam.thread)
            globalSpam.thread = nil
        end
        if globalSpam.spamBtnRef and globalSpam.spamBtnRef.Parent then
            globalSpam.spamBtnRef.Text = "Spam"
        end
        globalSpam.spamBtnRef  = nil
        globalSpam.animalName  = ""
    end
end

local function TriggerSpam(animalName, msValue, retryCount, spamBtnRef, jobId)
    StopCurrentSpam()
    globalSpam.active    = true
    globalSpam.animalName = animalName
    globalSpam.spamBtnRef = spamBtnRef
    if spamBtnRef then spamBtnRef.Text = "Stop" end

    local consoleLabel = AddConsoleEntry(animalName, msValue, 1, retryCount)

    globalSpam.thread = task.spawn(function()
        for i = 1, retryCount do
            if not globalSpam.active or globalSpam.animalName ~= animalName then break end
            if consoleLabel and consoleLabel.Parent then
                local timeStr = os.date("%H:%M")
                consoleLabel.Text = "[" .. timeStr .. "] [" .. i .. "/" .. retryCount .. "] " .. animalName .. " $" .. msValue .. "M/s"
            end
            if jobId and jobId ~= "" then
                pcall(function()
                    TeleportService:TeleportToPlaceInstance(game.PlaceId, jobId, player)
                end)
            end
            task.wait(1.5)
        end
        if globalSpam.active and globalSpam.animalName == animalName then
            StopCurrentSpam()
        end
    end)
end

-- WEBSOCKET CONNECT

local function processMessage(data)
    if data.type == "new_server" then
        local rawJobId = tostring(data.jobid or "")
        local jobId    = (rawJobId ~= "") and decryptJobId(rawJobId) or ""
        local name     = tostring(data.name     or "Unknown")
        local ms       = tostring(data.ms       or data.money or "0")
        local mutation = tostring(data.mutation or data.mut   or "Normal")
        if jobId ~= "" and logged_ids[jobId] then return end
        if jobId ~= "" then logged_ids[jobId] = true end
        _G.AddNotifierLog(name, ms, mutation, jobId)

    elseif data.type == "batch" and data.animals then
        local batch = {}
        for _, entry in ipairs(data.animals) do
            local rawJobId = tostring(entry.jobid or entry.jobId or "")
            local jobId    = (rawJobId ~= "") and decryptJobId(rawJobId) or ""
            if jobId == "" or not logged_ids[jobId] then
                if jobId ~= "" then logged_ids[jobId] = true end
                table.insert(batch, {
                    name  = tostring(entry.name     or "Unknown"),
                    ms    = tostring(entry.ms       or "0"),
                    mut   = tostring(entry.mutation or entry.mut or "Normal"),
                    jobId = jobId,
                })
            end
        end
        if #batch > 0 then _G.AddServerLog(batch) end
    end
end

local function Connect()
    if is_connecting then return end
    is_connecting = true
    SetStatus("CONNECTING")

    local ws = nil
    local success, result = pcall(function() return WebSocket.connect(WS_URL) end)

    if success and result then
        ws = result
        active_ws = ws
        SetStatus("CONNECTED")
        AddConsoleEntry("websocket", "ok", "connected", "live")

        ws.OnMessage:Connect(function(msg)
            local ok, data = pcall(function() return HttpService:JSONDecode(msg) end)
            if ok and data then processMessage(data) end
        end)

        ws.OnClose:Connect(function()
            active_ws = nil
            SetStatus("OFFLINE")
            AddConsoleEntry("websocket", "0", "closed", "retry")
        end)
    else
        active_ws = nil
        SetStatus("OFFLINE")
        AddConsoleEntry("websocket", "0", "failed", "err")
    end

    is_connecting = false
end

-- AUTO RECONNECT + HEARTBEAT
task.spawn(function()
    while true do
        if not active_ws then
            Connect()
        else
            local ok = pcall(function() active_ws:Send("ping") end)
            if not ok then
                active_ws = nil
                SetStatus("OFFLINE")
            end
        end
        task.wait(1.5)
    end
end)
